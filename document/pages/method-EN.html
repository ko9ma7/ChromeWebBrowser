  • OpenUrl
    Syntax: public void OpenUrl(string Url)
    public void OpenUrl(object request)
    Parameters: Url: Will open the specified website.
    request：In general, for the NewWindow event.
    Examples 1: browser.OpenUrl(newUrl);
    Examples 2: private void NewPage(string newUrl,object req)
    {
       TabPage newPage = new TabPage(newUrl);
       tabControl1.TabPages.Add(newPage);
       tabControl1.SelectTab(newPage);
       ChromeWebBrowser browser = new ChromeWebBrowser();
       browser.BrowserNewWindow += new NewWindowEventHandler(chromeWebBrowser1_BrowserNewWindow);
       browser.Initialize();
       newPage.Controls.Add(browser);
       browser.Validate();
       browser.Dock = DockStyle.Fill;
       if (newUrl.Contains("&"))
        browser.OpenUrl(req);
       else
        browser.OpenUrl(newUrl);
    }
    Remarks: Must be initialized before the browser to open.
